Python import streamlit as st import ccxt import pandas as pd import pandas_ta as ta import time # Configuração da Página st.set_page_config(page_title="Apex USDC Bot Analyzer", layout="wide") st.title("🚀 USDC Spot Pair Analyzer - High Frequency & Low Risk") exchange = ccxt.binance() def get_data(): # 1. Filtrar Mercados USDC markets = exchange.load_markets() symbols = [s for s in markets if '/USDC' in s and markets[s].get('spot')] analysis = [] # Barra de progresso progress_bar = st.progress(0) for i, symbol in enumerate(symbols[:20]): # Limitado aos 20 primeiros para performance try: # A. OHLCV para Indicadores (RSI e ATR) ohlcv = exchange.fetch_ohlcv(symbol, timeframe='1h', limit=50) df = pd.DataFrame(ohlcv, columns=['ts', 'o', 'h', 'l', 'c', 'v']) # Cálculo de Indicadores df['rsi'] = ta.rsi(df['c'], length=14) df['atr'] = ta.atr(df['h'], df['l'], df['c'], length=14) last_rsi = df['rsi'].iloc[-1] last_atr_pct = (df['atr'].iloc[-1] / df['c'].iloc[-1]) * 100 # B. Order Book (Profundidade/Spread) orderbook = exchange.fetch_order_book(symbol, limit=10) bid = orderbook['bids'][0][0] if orderbook['bids'] else 0 ask = orderbook['asks'][0][0] if orderbook['asks'] else 0 spread = ((ask - bid) / bid) * 100 if bid > 0 else 0 # C. Ticker (Volume e Trades) ticker = exchange.fetch_ticker(symbol) volume = ticker['quoteVolume'] # LÓGICA DE SCORE (Alta Volatilidade + Baixo Spread + RSI Saudável) # Penaliza moedas com RSI > 70 (risco de queda) ou RSI < 30 (baixa força) score = (volume * last_atr_pct) / (spread + 0.01) if last_rsi > 70 or last_rsi < 30: score = score * 0.5 analysis.append({ "Par": symbol, "Vol. 24h (USDC)": f"{volume:,.2f}", "RSI (1h)": round(last_rsi, 2), "Volatilidade (ATR %)": round(last_atr_pct, 2), "Spread (%)": round(spread, 4), "Score de Eficiência": round(score, 2), "Status": "🔥 Ideal" if (40 < last_rsi < 60 and spread < 0.1) else "⚠️ Risco/Lento" }) except Exception as e: continue progress_bar.progress((i + 1) / 20) return pd.DataFrame(analysis) # --- INTERFACE STREAMLIT --- if st.button('Atualizar Análise de Mercado'): with st.spinner('Analisando liquidez e indicadores...'): df_result = get_data() if not df_result.empty: df_sorted = df_result.sort_values(by="Score de Eficiência", ascending=False) # Métricas em destaque top_pair = df_sorted.iloc[0] col1, col2, col3 = st.columns(3) col1.metric("Melhor Par", top_pair['Par']) col2.metric("Menor Spread", f"{df_sorted['Spread (%)'].min()}%") col3.metric("Maior Volatilidade", f"{df_sorted['Volatilidade (ATR %)'].max()}%") # Tabela de Dados st.dataframe(df_sorted, use_container_width=True) st.success("Dica: Foque em pares com Spread < 0.05% e RSI entre 45-55 para Grid Trading.") else: st.error("Não foi possível coletar dados. Verifique sua conexão.") 3. Como essa App resolve seus problemas: Maior número de Trades (Filtro de ATR e Volume): A app não olha apenas o preço, ela olha o ATR %.
